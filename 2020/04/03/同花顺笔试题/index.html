题目一：读取文件重复行

import (
    "os"
    "bufio"
)

func getRepeatedLines(file string) map[string]bool {
    // file := "D:\\Projects\\leetcode\\example.log"

    if _, err := os.Stat(file); os.IsNotExist(err) {
        fmt.Println("File Is Not Existed.")
        return 
    }

    fd, err := os.Open(file)
    if err != nil {
        fmt.Println("Open File Failed.")
        return
    }

    repo := make(map[string]bool, 0)
    stream := bufio.NewReader(fd)
    for {
        line, err := stream.ReadBytes('\n')
        // fmt.Print(string(line))

        if _, ok := repo[string(line)]; !ok {
            repo[string(line)] = true
        }
        if err == io.EOF {
            fmt.Println("Read File Ended")
            break
        }
    }

    // fmt.Printf("%v", repo)
}

题目二：slice由哪几个部分构成？如下代码输出为？

s := []int{}
fmt.Println(s == nil)

slice结构体：

type slice struct {
    ptr *Elem   // 指向数组的指针
    len int     // 切片长度
    cap int     // 切片容量
}

输出：false
解释：s虽然被赋了个空值，但类型其实是个[]int类型，和nil无法比较。

题目三：反转整数数组的顺序

func revert(nums []int) []int {
    if len(nums) == 0 {
        return nums
    }

    for i := 0; i < len(nums)/2; i++ {
        nums[i], nums[len(nums)-1-i] = nums[len(nums)-1-i], nums[i]
    }
    return nums
}

题目五：想要并发输出0-9，以下代码是否存在问题，应该如何改进呢？

package main

import "fmt"

func main() {
    for i := 0; i < 10; i++ {
        go func() {
            fmt.Println(i)
        }()
    }
}
[solution 1]
func main() {
    wg := sync.WaitGroup{}
    wg.Add(10)

    for i := 0; i < 10; i++ 
        go func(i int) {
            fmt.Println("current: ", i)
            wg.Done()
        }(i)
    }
    wg.Wait()
}
[solution 2]
// 解法有问题，未能输出10个数
func main() {
    ch := make(chan int, 0)
    for i := 0; i < 10; i++ {
        go func(ch <-chan int) {
            i := <-ch
            fmt.Println(i)
            <-ch
        }(ch)

        ch <- i
    }
    close(ch)
}

题目七：修改以下代码，实现并发安全的cache机制，如果可以实现重复抑制（同一资源只有一个goroutine请求）可加分（不需要考虑cache更新问题和LRU）

package main

import (
    "io/ioutil"
    "net/http"
)

func main() {
    http.HandleFunc("/", func(w http.ResponseWriter, r *http.Request) {
        r.ParseForm()
        if len(r.Form["url"]) > 0 {
            body, _ := fetch(r.Fomr["url"][0])
            w.Writer(body)
        }
    })
    http.ListenAndServe(":8000", nil)
}

func fetch(url string) ([]byte, error) {
    resp, err := http.Get(url)
    if err != nil {
        return nil, err
    }
    defer resp.Close()
    return ioutil.ReadAll(resp.body)
}
